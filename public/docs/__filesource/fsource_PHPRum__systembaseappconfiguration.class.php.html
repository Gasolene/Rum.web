  477.             }
  478.         }

  494.         /**
  495.          * loads config data from an xml file
  496.          *
  497.          * @param   string    $file    path to config file
  498.          * @return  void 
  499.          */
  500.         public function loadAppConfig( $file )
  501.         {
  502.             // create xml parser resource
  503.             $xml_parser = xml_parser_create();
  504.  
  505.             if( file_exists( $file ))
  506.             {
  507.                 // open template
  508.                 $fp = fopen( $file, "r" );
  509.                 if(!($fp)) {
  510.                     throw new \System\Base\InvalidOperationException( 'could not open xml document' );
  511.                 }
  512.  
  513.                 // validate document
  514.                 \libxml_use_internal_errors(true);
  515.                 $dom = new \DOMDocument;
  516.                 $dom->load($file);
  517.                 if(!$dom->schemaValidate(__SYSTEM_PATH__.'/base/application.xsd')) {
  518.                     $errors = \libxml_get_errors();
  519.                     throw new \System\Base\InvalidOperationException( 'xml document does not validate: ' . $errors[0]->message . ' in ' . $errors[0]->file . ' on line ' . $errors[0]->line );
  520.                 }
  521.  
  522.                 // read xml data into array
  523.                 $array = array();
  524.                 while( $data = fread( $fp, filesize( $file ))) {
  525.                     $array = array();
  526.                     if( !xml_parse_into_struct( $xml_parser, $data, $array )) {
  527.                         throw new \System\Base\InvalidOperationException( sprintf(
  528.                             "Configuration XML Parse Error: %s at line %d",
  529.                             xml_error_string(xml_get_error_code( $xml_parser )),
  530.                             xml_get_current_line_number( $xml_parser )
  531.                             ));
  532.                     }
  533.                 }
  534.  
  535.                 // free resources
  536.                 xml_parser_free($xml_parser);
  537.  
  538.                 // parse node tree
  539.                 $index=0;
  540.                 $this->_getRootNode( $array, $index );
  541.             }
  542.         }
  543.  
  544.  
  545.         /**
  546.          * get root node
  547.          *
  548.          * @param   array    &$nodes    array of xml nodes
  549.          * @param   int        &$index    array index
  550.          * @return  void 
  551.          */
  552.         private function _getRootNode( &$nodes, &$index )
  553.         {
  554.             $node_data = $nodes[$index++];
  555.  
  556.             if( $node_data['tag'] === 'APPLICATION' &&
  557.                 $node_data['type'] != 'cdata' &&
  558.                 isset( $node_data['attributes'] )) {
  559.  
  560.                 if( $node_data['attributes']['STATE'] === 'on' ) {
  561.                     $this->state = \System\Base\AppState::On();
  562.                 }
  563.                 elseif( $node_data['attributes']['STATE'] === 'debug' ) {
  564.                     $this->state = \System\Base\AppState::Debug();
  565.                 }
  566.                 if( isset( $node_data['attributes']['LANG'] )) {
  567.                     $this->lang = $node_data['attributes']['LANG'];
  568.                 }
  569.                 if( isset( $node_data['attributes']['CHARSET'] )) {
  570.                     $this->charset = $node_data['attributes']['CHARSET'];
  571.                 }
  572.  
  573.                 if( $node_data['type'] === 'open' )
  574.                 {
  575.                     $this->_getNodes( $nodes, $index );
  576.                 }
  577.             }
  578.  
  579.             return;
  580.         }
  581.  
  582.  
  583.         /**
  584.          * get all nodes
  585.          *
  586.          * @param   array    &$nodes    array of xml nodes
  587.          * @param   int        &$index    array index
  588.          * @return  void 
  589.          */
  590.         private function _getNodes( &$nodes, &$index )
  591.         {
  592.             while( $index < sizeof( $nodes ))
  593.             {
  594.                 $node_data = $nodes[$index++];
  595.  
  596.                 if( $node_data['type'] === 'close' )
  597.                 {
  598.                     return;
  599.                 }
  600.                 // app-settings
  601.                 elseif( $node_data['tag'] === 'APP-SETTINGS' && $node_data['type'] != 'cdata' )
  602.                 {
  603.                     if( $node_data['type'] === 'open' )
  604.                     {
  605.                         $this->_getAppSettingNodes( $nodes, $index );
  606.                     }
  607.                 }
  608.                 // pages
  609.                 elseif( $node_data['tag'] === 'PAGES' && $node_data['type'] != 'cdata' && isset( $node_data['attributes'] ))
  610.                 {
  611.                     if( isset( $node_data['attributes']['THEME'] )) {
  612.                         // TODO: Deprecated
  613.                         $this->defaultTheme = $node_data['attributes']['THEME'];
  614.                     }
  615.                     if( isset( $node_data['attributes']['DEFAULT-THEME'] )) {
  616.                         $this->defaultTheme = $node_data['attributes']['DEFAULT-THEME'];
  617.                     }
  618.  
  619.                     $this->_closeNode( $nodes, $index );
  620.                 }
  621.                 // viewstate
  622.                 elseif( $node_data['tag'] === 'VIEWSTATE' && $node_data['type'] != 'cdata' && isset( $node_data['attributes'] ))
  623.                 {
  624.                     if( isset( $node_data['attributes']['ENABLED'] )) {
  625.                         if( strtolower( $node_data['attributes']['ENABLED'] === 'true' )) {
  626.                             $this->viewStateEnabled = true;
  627.                         }
  628.                         else {
  629.                             $this->viewStateEnabled = false;
  630.                         }
  631.                     }
  632.                     if( isset( $node_data['attributes']['METHOD'] )) {
  633.                         $this->viewStateMethod = $node_data['attributes']['METHOD'];
  634.                     }
  635.                     if( isset( $node_data['attributes']['EXPIRES'] )) {
  636.                         if( is_numeric( $node_data['attributes']['EXPIRES'] )) {
  637.                             $this->viewStateExpires = (int) $node_data['attributes']['EXPIRES'];
  638.                         }
  639.                     }
  640.  
  641.                     $this->_closeNode( $nodes, $index );
  642.                 }
  643.                 // request
  644.                 elseif( $node_data['tag'] === 'REQUEST' && $node_data['type'] != 'cdata' && isset( $node_data['attributes'] ))
  645.                 {
  646.                     if( isset( $node_data['attributes']['DEFAULT'] )) {
  647.                         $this->defaultController = $node_data['attributes']['DEFAULT'];
  648.                     }
  649.                     if( isset( $node_data['attributes']['PARAM'] )) {
  650.                         $this->requestParameter = $node_data['attributes']['PARAM'];
  651.                     }
  652.                     if( isset( $node_data['attributes']['FRIENDLY-URIS'] )) {
  653.                         if( strtolower( $node_data['attributes']['FRIENDLY-URIS'] === 'true' )) {
  654.                             $this->rewriteURIS = true;
  655.                         }
  656.                         else {
  657.                             $this->rewriteURIS = false;
  658.                         }
  659.                     }
  660.  
  661.                     $this->_closeNode( $nodes, $index );
  662.                 }
  663.                 // authentication
  664.                 elseif( $node_data['tag'] === 'AUTHENTICATION' && $node_data['type'] != 'cdata' )
  665.                 {
  666.                     if( isset( $node_data['attributes'] )) {
  667.                         // authenticationMethod
  668.                         if( isset( $node_data['attributes']['METHOD'] )) {
  669.                             $this->authenticationMethod = strtolower( $node_data['attributes']['METHOD'] );
  670.                         }
  671.                         // authenticationDeny
  672.                         if( isset( $node_data['attributes']['DENY'] )) {
  673.                             $this->authenticationDeny = explode( ',', strtolower( $node_data['attributes']['DENY'] ));
  674.                         }
  675.                         // authenticationAllow
  676.                         if( isset( $node_data['attributes']['ALLOW'] )) {
  677.                             $this->authenticationAllow = explode( ',', strtolower( $node_data['attributes']['ALLOW'] ));
  678.                         }
  679.                         // authenticationRestrict
  680.                         if( isset( $node_data['attributes']['RESTRICT'] )) {
  681.                             $this->authenticationRestrict = (int)$node_data['attributes']['RESTRICT'];
  682.                         }
  683.                         // authenticationMaxInvalidAttempts
  684.                         if( isset( $node_data['attributes']['MAXINVALIDATTEMPTS'] )) {
  685.                             $this->authenticationMaxInvalidAttempts = (int)$node_data['attributes']['MAXINVALIDATTEMPTS'];
  686.                         }
  687.                         // authenticationAttemtpWindow
  688.                         if( isset( $node_data['attributes']['ATTEMPTWINDOW'] )) {
  689.                             $this->authenticationAttemtpWindow = (int) $node_data['attributes']['ATTEMPTWINDOW'];
  690.                         }
  691.                         // authenticationRequireSSL
  692.                         if( isset( $node_data['attributes']['REQUIRESSL'] )) {
  693.                             if( strtolower( $node_data['attributes']['REQUIRESSL'] === 'true' )) {
  694.                                 $this->authenticationRequireSSL = true;
  695.                             }
  696.                             else {
  697.                                 $this->authenticationRequireSSL = false;
  698.                             }
  699.                         }
  700.                     }
  701.  
  702.                     if( $node_data['type'] === 'open' )
  703.                     {
  704.                         $this->_getAuthenticationNodes( $nodes, $index );
  705.                     }
  706.                 }
  707.                 // authorization
  708.                 elseif( $node_data['tag'] === 'AUTHORIZATION' && $node_data['type'] != 'cdata' )
  709.                 {
  710.                     if( isset( $node_data['attributes'] )) {
  711.                         // deny
  712.                         if( isset( $node_data['attributes']['DENY'] )) {
  713.                             $this->authorizationDeny = array();
  714.                             $roles = explode( ',', $node_data['attributes']['DENY'] );
  715.                             foreach($roles as $role)
  716.                             {
  717.                                 if(trim($role)) $this->authorizationDeny[] = trim($role);
  718.                             }
  719.                         }
  720.                         // allow
  721.                         if( isset( $node_data['attributes']['ALLOW'] )) {
  722.                             $this->authorizationAllow = array();
  723.                             $roles = explode( ',', $node_data['attributes']['ALLOW'] );
  724.                             foreach($roles as $role)
  725.                             {
  726.                                 if(trim($role)) $this->authorizationAllow[] = trim($role);
  727.                             }
  728.                         }
  729.                         // restrict
  730.                         if( isset( $node_data['attributes']['RESTRICT'] )) {
  731.                             $this->authorizationRestrict = (int)$node_data['attributes']['RESTRICT'];
  732.                         }
  733.                         // requireSSL
  734.                         if( isset( $node_data['attributes']['REQUIRESSL'] )) {
  735.                             if( strtolower( $node_data['attributes']['REQUIRESSL'] === 'true' )) {
  736.                                 $this->authorizationRequireSSL = true;
  737.                             }
  738.                             else {
  739.                                 $this->authorizationRequireSSL = false;
  740.                             }
  741.                         }
  742.                     }
  743.  
  744.                     if( $node_data['type'] === 'open' )
  745.                     {
  746.                         $this->_getAuthorizationNodes( $nodes, $index );
  747.                     }
  748.                 }
  749.                 // session
  750.                 elseif( $node_data['tag'] === 'SESSION' && $node_data['type'] != 'cdata' && isset( $node_data['attributes'] ))
  751.                 {
  752.                     if( isset( $node_data['attributes']['COOKIELESS'] )) {
  753.                         if( strtolower( $node_data['attributes']['COOKIELESS'] === 'true' )) {
  754.                             $this->cookielessSession = true;
  755.                         }
  756.                         else {
  757.                             $this->cookielessSession = false;
  758.                         }
  759.                     }
  760.                     if( isset( $node_data['attributes']['TIMEOUT'] )) {
  761.                         if( is_numeric( $node_data['attributes']['TIMEOUT'] )) {
  762.                             $this->sessionTimeout = (int) $node_data['attributes']['TIMEOUT'];
  763.                         }
  764.                     }
  765.  
  766.                     $this->_closeNode( $nodes, $index );
  767.                 }
  768.                 // cache
  769.                 elseif( $node_data['tag'] === 'CACHE' && $node_data['type'] != 'cdata' && isset( $node_data['attributes'] ))
  770.                 {
  771.                     if( isset( $node_data['attributes']['ENABLED'] )) {
  772.                         if( strtolower( $node_data['attributes']['ENABLED'] === 'true' )) {
  773.                             $this->cacheEnabled = true;
  774.                         }
  775.                         else {
  776.                             $this->cacheEnabled = false;
  777.                         }
  778.                     }
  779.                     if( isset( $node_data['attributes']['EXPIRES'] )) {
  780.                         if( is_numeric( $node_data['attributes']['EXPIRES'] )) {
  781.                             $this->cacheExpires = (int) $node_data['attributes']['EXPIRES'];
  782.                         }
  783.                     }
  784.  
  785.                     $this->_closeNode( $nodes, $index );
  786.                 }
  787.                 // data-source
  788.                 elseif( $node_data['tag'] === 'DATA-SOURCE' && $node_data['type'] != 'cdata' && isset( $node_data['attributes'] ))
  789.                 {
  790.                     if( isset( $node_data['attributes']['DSN'] )) {
  791.                         $this->dsn = $node_data['attributes']['DSN'];
  792.                     }
  793.  
  794.                     $this->_closeNode( $nodes, $index );
  795.                 }
  796.                 // errors
  797.                 elseif( $node_data['tag'] === 'ERRORS' && $node_data['type'] != 'cdata' )
  798.                 {
  799.                     if( $node_data['type'] === 'open' )
  800.                     {
  801.                         $this->_getErrorHandlingNodes( $nodes, $index );
  802.                     }
  803.                 }
  804.             }
  805.         }
  806.  
  807.  
  808.         /**
  809.          * get app-setting nodes
  810.          *
  811.          * @param   array    &$nodes    array of xml nodes
  812.          * @param   int        &$index    array index
  813.          * @return  void 
  814.          */
  815.         private function _getAppSettingNodes( &$nodes, &$index )
  816.         {
  817.             while( $index < sizeof( $nodes ))
  818.             {
  819.                 $node_data = $nodes[$index++];
  820.  
  821.                 if( $node_data['type'] === 'close' )
  822.                 {
  823.                     return;
  824.                 }
  825.                 elseif( $node_data['tag'] === 'ADD' &&
  826.                         $node_data['type'] != 'cdata' &&
  827.                         isset( $node_data['attributes'] ))
  828.                 {
  829.                     // error page
  830.                     if( isset( $node_data['attributes']['KEY'] ) &&
  831.                         isset( $node_data['attributes']['VALUE'] ))
  832.                     {
  833.                         $this->appsettings[$node_data['attributes']['KEY']] = $node_data['attributes']['VALUE'];
  834.                     }
  835.  
  836.                     $this->_closeNode( $nodes, $index );
  837.                 }
  838.             }
  839.         }
  840.  
  841.  
  842.         /**
  843.          * get authentication Nodes
  844.          *
  845.          * @param   array    &$nodes    array of xml nodes
  846.          * @param   int        &$index    array index
  847.          * @return  void 
  848.          */
  849.         private function _getAuthenticationNodes( &$nodes, &$index )
  850.         {
  851.             while( $index < sizeof( $nodes ))
  852.             {
  853.                 $node_data = $nodes[$index++];
  854.  
  855.                 if( $node_data['type'] === 'close' )
  856.                 {
  857.                     return;
  858.                 }
  859.                 // basic
  860.                 elseif( $node_data['tag'] === 'BASIC' &&
  861.                         $node_data['type'] != 'cdata' &&
  862.                         isset( $node_data['attributes'] ))
  863.                 {
  864.                     // authenticationBasicRealm
  865.                     if( isset( $node_data['attributes']['REALM'] )) {
  866.                         $this->authenticationBasicRealm = $node_data['attributes']['REALM'];
  867.                     }
  868.  
  869.                     $this->_closeNode( $nodes, $index );
  870.                 }
  871.                 // forms
  872.                 elseif( $node_data['tag'] === 'FORMS' &&
  873.                         $node_data['type'] != 'cdata' &&
  874.                         isset( $node_data['attributes'] ))
  875.                 {
  876.                     // authenticationFormsLoginPage
  877.                     if( isset( $node_data['attributes']['LOGINPAGE'] )) {
  878.                         $this->authenticationFormsLoginPage = $node_data['attributes']['LOGINPAGE'];
  879.                     }
  880.                     // authenticationFormsCookieName
  881.                     if( isset( $node_data['attributes']['COOKIENAME'] )) {
  882.                         $this->authenticationFormsCookieName = $node_data['attributes']['COOKIENAME'];
  883.                     }
  884.                     // authenticationFormsSecret
  885.                     if( isset( $node_data['attributes']['SECRET'] )) {
  886.                         $this->authenticationFormsSecret = $node_data['attributes']['SECRET'];
