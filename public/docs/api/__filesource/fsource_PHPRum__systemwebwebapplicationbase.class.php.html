  491.         {

  519.  
  520.  
  521.         /**
  522.          * Processe HTTP Request and create actions to handle business logic.  Then render the view.
  523.          * Can be overwridden to provide additional or alternaitive private functionality.
  524.          *
  525.          * @param   HTTPRequest        $request    HTTPRequest object
  526.          * @return  void 
  527.          */
  528.         protected function requestProcessor( \System\Web\HTTPRequest &$request )
  529.         {
  530.             // get controller from request
  531.             if( isset( $request->post[$this->config->requestParameter] ))
  532.             {
  533.                 // Get Controller based on HTTP POST page parameter
  534.                 $this->forwardPage = $request->post[$this->config->requestParameter];
  535.             }
  536.             elseif( isset( $request->get[$this->config->requestParameter] ))
  537.             {
  538.                 // Get Controller based on HTTP GET page parameter
  539.                 $this->forwardPage = $request->get[$this->config->requestParameter];
  540.             }
  541.             else
  542.             {
  543.                 // Get Controller based on XML configuration
  544.                 $this->forwardPage = $this->config->defaultController;
  545.             }
  546.  
  547.             if( isset( $request->request['lang'] ))
  548.             {
  549.                 // Get Language
  550.                 $this->lang = $request->request['lang'];
  551.             }
  552.  
  553.             $request[$this->config->requestParameter] = strtolower( $this->forwardPage );
  554.  
  555.             while( $this->forwardPage )
  556.             {
  557.                 // Dispatch controller
  558.                 $requestHandler = $this->getRequestHandler( $this->forwardPage );
  559.  
  560.                 // raise event
  561.                 $this->events->raise(new Events\WebApplicationHandleRequestEvent(), $this);
  562.  
  563.                 // Authentication/Authorization
  564.                 if(\System\Security\Authentication::authenticated())
  565.                 {
  566.                     if(\System\Security\Authentication::authorized())
  567.                     {
  568.                         $headersCacheId = '';
  569.                         $outputCacheId = '';
  570.  
  571.                         // Output Caching
  572.                         if( $requestHandler->outputCache <> 0 )
  573.                         {
  574.                             $hashId = 'hash:' . $requestHandler->controllerId;
  575.  
  576.                             // Output Caching Enabled
  577.                             if( \System\Web\HTTPRequest::getRequestMethod() === 'GET' )
  578.                             {
  579.                                 $hash = \System\Web\WebApplicationBase::getInstance()->cache->get($hashId);
  580.  
  581.                                 if(!$hash)
  582.                                 {
  583.                                     $hash = $requestHandler->controllerId;
  584.  
  585.                                     if( $this->config->cacheEnabled )
  586.                                     {
  587.                                         \System\Web\WebApplicationBase::getInstance()->cache->put($hashId, $hash, $requestHandler->outputCache);
  588.                                     }
  589.                                 }
  590.  
  591.                                 // Generate Cache ID's
  592.                                 $headersCacheId = 'output_headers:'.$hash.$requestHandler->getCacheId();
  593.                                 $outputCacheId = 'output:'.$hash.$requestHandler->getCacheId();
  594.  
  595.                                 // Output Cache Exists
  596.                                 if( \System\Web\WebApplicationBase::getInstance()->cache->get( $outputCacheId ))
  597.                                 {
  598.                                     $response = new \System\Web\HTTPResponse();
  599.                                     $headers = \System\Web\WebApplicationBase::getInstance()->cache->get($headersCacheId);
  600.  
  601.                                     // Send Headers
  602.                                     if( $headers )
  603.                                     {
  604.                                         foreach( $headers as $header )
  605.                                         {
  606.                                             \System\Web\HTTPResponse::addHeader($header);
  607.                                         }
  608.                                     }
  609.  
  610.                                     // Send Output
  611.                                     \System\Web\HTTPResponse::write( \System\Web\WebApplicationBase::getInstance()->cache->get( $outputCacheId ));
  612.  
  613.                                     $this->insertDebug($response, $requestHandler->outputCache );
  614.  
  615.                                     return;
  616.                                 }
  617.                             }
  618.                             else
  619.                             {
  620.                                 \System\Web\WebApplicationBase::getInstance()->cache->clear($hashId);
  621.                             }
  622.                         }
  623.  
  624.                         // get View
  625.                         $view = $requestHandler->getView( $request );
  626.  
  627.                         if( !$this->forwardPage )
  628.                         {
  629.                             if( $this->forwardURI )
  630.                             {
  631.                                 // Forward URI (redirect)
  632.                                 $this->forwardURI( $this->forwardURI, (array)$this->forwardParams );
  633.                                 return;
  634.                             }
  635.  
  636.                             // Render View
  637.                             $response = new \System\Web\HTTPResponse(); // start output buffer
  638.                             $view->render();
  639.  
  640.                             // Output Caching Capture
  641.                             if( $requestHandler->outputCache > 0 &&
  642.                                 \System\Web\HTTPRequest::getRequestMethod() === 'GET' &&
  643.                                 $this->messages->count === 0 &&
  644.                                 $this->config->cacheEnabled )
  645.                             {
  646.                                 \System\Web\WebApplicationBase::getInstance()->cache->put( $headersCacheId, \System\Web\HTTPResponse::getResponseHeaders(), $requestHandler->outputCache );
  647.                                 \System\Web\WebApplicationBase::getInstance()->cache->put( $outputCacheId, \System\Web\HTTPResponse::getResponseContent(), $requestHandler->outputCache );
  648.                             }
  649.  
  650.                             // Debug
  651.                             $this->insertDebug($response);
  652.  
  653.                             // Cleanup
  654.                             $this->messages->removeAll();
  655.                             $this->prevPage = '';
  656.                         }
  657.                         else
  658.                         {
  659.                             // Forward Request (redirect)
  660.                             foreach( $this->forwardParams as $key => $value )
  661.                             {
  662.                                 $request[$key] = $value;
  663.                             }
  664.                             $this->forwardParams = array();
  665.                         }
  666.  
  667.                         // Remove reference
  668.                         unset($this->requestHandler);
  669.                     }
  670.                     else
  671.                     {
  672.                         // Not Authorized
  673.                         WebApplicationBase::getInstance()->sendHTTPError(401);
  674.                     }
  675.                 }
  676.                 else
  677.                 {
  678.                     // Not Authenticated
  679.                     \System\Security\Authentication::redirectToLogin();
  680.                 }
  681.             }
  682.         }

  684.  
  685.         /**
  686.          * event triggered by an uncaught Exception thrown in the application, can be overridden to provide error handling.
  687.          *
  688.          * @param  Exception    $e 
  689.          *
  690.          * @return void 
  691.          */
  692.         protected function handleException(\Exception $e)
  693.         {
  694.             if( $this->debug )
  695.             {
  696.                 $backtrace = $e->getTrace();
  697.  
  698.                 foreach( $backtrace as $trace )
  699.                 {
  700.                     if( strstr( $trace['file'], __SYSTEM_PATH__ ) === false )
  701.                     {
  702.                         $file = $trace['file'];
  703.                         $line = $trace['line'];
  704.                         break;
  705.                     }
  706.                 }
  707.  
  708.                 $source = '';
  709.                 $line = isset($line)?$line:$e->getLine();
  710.                 $file = isset($file)?$file:$e->getFile();
  711.                 $contents = file($file);
  712.                 $filename = str_replace( __ROOT__, '', $file );
  713.  
  714.                 for( $i = $line > 3 ? $line - 3 : 0, $ii = 0; $ii <= 5; $ii++, $i++ )
  715.                 {

  718.  
  719.                     if( isset( $contents[$i-1] ))
  720.                     {
  721.                         $contents[$i-1] = \Rum::escape($contents[$i-1]);
  722.  
  723.                         if( $i === $line )
  724.                         {
  725.                             $source .= "<span style=\"font-weight:bold;color:#AA0000\">Line $current_line{$contents[$i-1]}</span>";
  726.                         }
  727.                         else
  728.                         {
  729.                             $source .= "<span>Line $current_line{$contents[$i-1]}</span>";
  730.                         }
  731.                     }
  732.                 }
  733.  
  734.                 // bad form, not structured - but the only way to clear the output buffer
  735.                 @ob_clean();
  736.  
  737.                 $response = new \System\Web\HTTPResponse();
  738.  
  739.                 \System\Web\HTTPResponse::addHeader( "Content-Type: text/html" );
  740.                 \System\Web\HTTPResponse::write( "<!DOCTYPE html>
  741. <html lang=\"en\">
  742. <head>
  743. <title>Unhandled Exception: ".htmlentities($e->getMessage())."</title>
  744. <meta http-equiv=\"Content-Type\" content=\"text/html;charset=utf-8\">
  745. <link href=\"" . $this->config->assets . "/web/exception.css\" rel=\"stylesheet\" type=\"text/css\" media=\"all\" />
  746. <link href=\"" . $this->config->assets . "/web/debug.css\" rel=\"stylesheet\" type=\"text/css\" media=\"all\" />
  747. <script src=\"" . $this->config->assets . "/web/debug.js\" type=\"text/javascript\"></script>
  748. </head>
  749. <body>
  750.  
  751. <div id=\"tl\">
  752. <div id=\"tr\">
  753. <div id=\"bl\">
  754. <div id=\"br\">
  755.  
  756. <h1>Unhandled Exception in " . strrchr( $e->getFile(), '/' ) . "</h1>
  757.  
  758. <div id=\"details\" class=\"fail\">
  759. <h2>Runtime Error: ".htmlentities($e->getMessage())."</h2>
  760. <p><strong>Description:</strong> An unhandled exception occurred during execution</p>
  761. <p><strong>Details:</strong> " . get_class($e) . ": {$e->getMessage()}</p>
  762. <p><strong>Source File:</strong> {$filename} <strong>on line:</strong> {$line}
  763. </div>
  764.  
  765. <pre>
  766. <strong>Source Information:</strong>
  767.  
  768. {$source}
  769. </pre>
  770.  
  771. <pre>
  772. <strong>Stack Trace:</strong>\r\n\r\n");
  773.  
  774.                 $this->dumpCallStack($e->getTrace());
  775.                 \System\Web\HTTPResponse::write( "
  776. </pre>
  777.  
  778. <!--<p class=\"dump\" id=\"debug_show\"><a href=\"#debug_dump\" onclick=\"document.getElementById('debug_info').style.display='block';document.getElementById('debug_show').style.display='none';\">Show Debug Information</a></p>-->
  779. <div style=\"display:none;\" id=\"debug_info\">");
  780.  
  781.                 $this->dumpDebug();
  782.  
  783.                 \System\Web\HTTPResponse::write( "
  784. </div>
  785.  
  786. <div id=\"version\">
  787. <p><strong>Framework Version:</strong> ".\System\Base\FRAMEWORK_VERSION_STRING."</p>

  789.

  791. </div>
  792. </div>
  793. </div>
  794.  
  795. </body>
  796. </html>" );
  797.  
  798.                 // handle exceptions on ajax requests
  799.                 if($this->requestHandler)
  800.                 {
  801.                     if($this->requestHandler instanceof PageControllerBase)
  802.                     {
  803.                         if($this->requestHandler->isAjaxPostBack)
  804.                         {
  805.                             $content = \System\Web\HTTPResponse::getResponseContent();
  806.                             \System\Web\HTTPResponse::clear();
  807.                             \System\Web\HTTPResponse::write( "
  808. ExceptionWindow=window.open('', 'Dialog', 'height=800,width=1024,toolbar=no,scrollbars=yes,menubar=no,directories=no,location=no,status=no');
  809. ExceptionWindow.document.write(\"".addslashes(str_replace(array("\r\n", "\r", "\n"), '', $content))."\")");
  810.                         }
  811.                     }
  812.                 }
  813.             }
  814.             else
  815.             {
  816.                 $this->logger->log( "Uncaught Exception: {$e->getMessage()} in {$e->getFile()} on line {$e->getLine()}", 'error' );
  817.                 \System\Web\WebApplicationBase::getInstance()->sendHTTPError( 500 );
  818.             }
  819.         }
  820.  
  821.  
  822.         /**
  823.          * event triggered by an error in the application, can be overridden to provide error handling.
  824.          *
  825.          * @param  string    $errno        error code
  826.          * @param  string    $errstr        error description
  827.          * @param  string    $errfile    file

  834.                E_ERROR                => "Error",
  835.                E_WARNING            => "Warning",
  836.                E_PARSE                => "Parsing Error",
  837.                E_NOTICE                => "Notice",

  839.                E_CORE_ERROR            => "Core Error",
  840.                E_CORE_WARNING        => "Core Warning",

  869.             }
  870.         }

  884.  
  885.  
  886.         /**
  887.          * Start session, sets session id (using request parameter)

  890.          */
  891.         private function startSession()
  892.         {
  893.             if( ApplicationBase::getInstance()->config->cookielessSession )
  894.             {
  895.                 if( isset( HTTPRequest::$post['PHPSESSID'] ))
  896.                 {
  897.                     $this->session->start( HTTPRequest::$post['PHPSESSID'] );
  898.                     unset( HTTPRequest::$post['PHPSESSID'] );
  899.                 }
  900.                 elseif( isset( HTTPRequest::$get['PHPSESSID'] ))
  901.                 {
  902.                     $this->session->start( HTTPRequest::$get['PHPSESSID'] );
  903.                     unset( HTTPRequest::$get['PHPSESSID'] );
  904.                 }
  905.                 else
  906.                 {
  907.                     $this->session->start();
  908.                 }
  909.             }
  910.             else
  911.             {
  912.                 $this->session->start();
  913.             }
  914.         }

  916.

  922.         private function endSession()

  926.  
  927.  
  928.         /**
  929.          * run helper commands
  930.          *
  931.          * @param   HTTPRequest        $request        HTTPRequest object
  932.          * @return  void 
  933.          */
  934.         private function handleRequestParams( \System\Web\HTTPRequest &$request )
  935.         {
  936.             if(isset($request->get["page"]) && isset($request->get["id"]))
  937.             {
  938.                 if($_SERVER[__ENV_PARAMETER__]==__DEV_ENV__||$_SERVER[__ENV_PARAMETER__]==__TEST_ENV__)
  939.                 {
  940.                     if($request->get["page"]=='dev' && ($request->get["id"]=="clean" || $request->get["id"]=="build"))
  941.                     {
